<objective>
Implement per-file upload TTLs with a SQLite tracking table, background auto-deletion sweep, WebSocket toast notifications on deletion, countdown expiry badges in the file listing, boot-time cleanup of expired drop box files, and agent-side expired file prompting on mount reconnect.

Purpose: Files uploaded to the drop box should not accumulate forever. Per-file TTLs let users choose how long their files persist (default 1 day). Expired files are auto-deleted with browser notifications, keeping the drop box clean without manual intervention. For agent-backed mounts, expired files are communicated to the agent on reconnect so the user can decide to keep or delete them.
Output: file_ttl SQLite table, file_ttl_sweep background task, TTL query param on upload, ExpiryBadge component, WebSocket toast on deletion, boot-time cleanup, agent-side expired file prompt.
</objective>

1. Create `relay/app/services/file_ttl_db.py`:
   ```python
   """File TTL database operations -- tracks per-file expiry metadata in SQLite."""
   import time
   import aiosqlite

   _CREATE_FILE_TTL_TABLE = """
   CREATE TABLE IF NOT EXISTS file_ttl (
       id INTEGER PRIMARY KEY AUTOINCREMENT,
       mount_code TEXT NOT NULL,
       file_path TEXT NOT NULL,
       expires_at REAL NOT NULL,
       created_at REAL NOT NULL,
       UNIQUE(mount_code, file_path)
   )
   """
   _CREATE_FILE_TTL_INDEX = """
   CREATE INDEX IF NOT EXISTS idx_file_ttl_expires ON file_ttl (expires_at)
   """

   class FileTtlDb:
       def __init__(self, db: aiosqlite.Connection) -> None:
           self._db = db

       async def init_table(self) -> None:
           await self._db.execute(_CREATE_FILE_TTL_TABLE)
           await self._db.execute(_CREATE_FILE_TTL_INDEX)
           await self._db.commit()

       async def record_file_ttl(self, mount_code: str, file_path: str, ttl_seconds: int) -> None:
           now = time.time()
           expires_at = now + ttl_seconds
           await self._db.execute(
               "INSERT OR REPLACE INTO file_ttl (mount_code, file_path, expires_at, created_at) "
               "VALUES (?, ?, ?, ?)",
               (mount_code, file_path, expires_at, now),
           )
           await self._db.commit()

       async def get_expired(self) -> list[tuple[str, str, float]]:
           """Return (mount_code, file_path, expires_at) for all expired records."""
           now = time.time()
           async with self._db.execute(
               "SELECT mount_code, file_path, expires_at FROM file_ttl WHERE expires_at <= ?",
               (now,),
           ) as cursor:
               return await cursor.fetchall()

       async def get_expired_for_mount(self, mount_code: str) -> list[tuple[str, float]]:
           """Return (file_path, expires_at) for expired records in a specific mount."""
           now = time.time()
           async with self._db.execute(
               "SELECT file_path, expires_at FROM file_ttl WHERE mount_code = ? AND expires_at <= ?",
               (mount_code, now),
           ) as cursor:
               return await cursor.fetchall()

       async def delete_record(self, mount_code: str, file_path: str) -> None:
           await self._db.execute(
               "DELETE FROM file_ttl WHERE mount_code = ? AND file_path = ?",
               (mount_code, file_path),
           )
           await self._db.commit()

       async def get_ttl_for_mount(self, mount_code: str) -> list[tuple[str, float]]:
           """Return (file_path, expires_at) for all files in a mount with TTL."""
           async with self._db.execute(
               "SELECT file_path, expires_at FROM file_ttl WHERE mount_code = ?",
               (mount_code,),
           ) as cursor:
               return await cursor.fetchall()

       async def delete_expired_for_mount(self, mount_code: str) -> list[str]:
           """Delete expired records for a mount. Returns list of deleted file paths."""
           now = time.time()
           async with self._db.execute(
               "SELECT file_path FROM file_ttl WHERE mount_code = ? AND expires_at <= ?",
               (mount_code, now),
           ) as cursor:
               rows = await cursor.fetchall()
           paths = [row[0] for row in rows]
           if paths:
               await self._db.execute(
                   "DELETE FROM file_ttl WHERE mount_code = ? AND expires_at <= ?",
                   (mount_code, now),
               )
               await self._db.commit()
           return paths
   ```

2. Create `relay/app/services/file_ttl_sweep.py`:
   ```python
   """Background file TTL sweep -- deletes expired files and broadcasts WebSocket notifications."""
   import asyncio, logging, os
   from pathlib import Path
   from relay.app.config import RelayConfig
   from relay.app.services.file_ttl_db import FileTtlDb

   logger = logging.getLogger("relay.file_ttl_sweep")

   async def file_ttl_sweep_once(
       file_ttl_db: FileTtlDb,
       data_dir: Path,
       dropbox_code: str,
       broadcast_fn: ... | None,  # Callable for WebSocket broadcast
   ) -> list[str]:
       """Single sweep iteration. Returns list of deleted file paths."""
       expired = await file_ttl_db.get_expired()
       deleted: list[str] = []
       for mount_code, file_path, expires_at in expired:
           # Only delete files for the drop box (relay has filesystem access)
           if mount_code == dropbox_code:
               full_path = data_dir / "dropbox" / file_path.lstrip("/")
               if full_path.exists():
                   try:
                       os.remove(full_path)
                       logger.info("Deleted expired file: %s (mount=%s)", file_path, mount_code)
                       deleted.append(file_path)
                   except OSError:
                       logger.exception("Failed to delete expired file: %s", file_path)
           await file_ttl_db.delete_record(mount_code, file_path)
       # Broadcast toast for each deleted file
       if broadcast_fn is not None and deleted:
           for fp in deleted:
               await broadcast_fn({"type": "toast", "message": f"File expired and removed: {Path(fp).name}"})
       return deleted

   async def run_file_ttl_sweep(
       file_ttl_db: FileTtlDb,
       data_dir: Path,
       dropbox_code: str,
       sweep_interval: int,
       broadcast_fn: ... | None,
   ) -> None:
       while True:
           await asyncio.sleep(sweep_interval)
           await file_ttl_sweep_once(file_ttl_db, data_dir, dropbox_code, broadcast_fn)
   ```
   Use a configurable sweep interval. A reasonable default is 60 seconds (Claude's discretion per CONTEXT.md).

   ```python
   @router.get("/api/files", dependencies=[Depends(require_full_access)])
   async def get_files(path: str = Query("")) -> dict:
       config = get_server_config()
       try:
           listing = list_directory(config.shared_folder, path)
           result = listing.model_dump()

           # Enrich file entries with TTL expiry timestamps
           try:
               from relay.app.services.file_ttl_db import get_file_ttl_db
               if config.mount_code is not None:
                   file_ttl_db = get_file_ttl_db()
                   ttl_records = await file_ttl_db.get_ttl_for_mount(config.mount_code)
                   # Build lookup: relative_path -> expires_at (ISO string)
                   from datetime import datetime, timezone
                   ttl_map: dict[str, str] = {}
                   for file_path, expires_at in ttl_records:
                       ttl_map[file_path] = datetime.fromtimestamp(expires_at, tz=timezone.utc).isoformat()
                   # Merge into file entries
                   for entry in result.get("entries", []):
                       entry["expires_at"] = ttl_map.get(entry.get("path", ""), None)
           except (RuntimeError, ImportError):
               # Standalone mode -- no TTL tracking, set all to null
               for entry in result.get("entries", []):
                   entry["expires_at"] = None

           return result
       except PathTraversalError as exc:
           return _handle_path_traversal(exc)
       except FileNotFoundError as exc:
           return _handle_not_found(exc)
   ```
